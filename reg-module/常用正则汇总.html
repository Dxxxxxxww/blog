# 常用正则汇总

  1. 整数，小数，正负数校验时（不能用在输入时，无法解决单个 ‘-’ 问题）
const reg = /^(-)?\d+(\.\d{1,2})?$/

console.log(reg.test('')) // false

console.log(reg.test('-')) // false
 
console.log(reg.test('-0')) // true
 
console.log(reg.test('-0.1')) // true
 
console.log(reg.test('-0.12')) // true
 
console.log(reg.test('-10.12')) // true
 
console.log(reg.test('10.12')) // true
 
console.log(reg.test('0.12')) // true
 
console.log(reg.test('0.1')) // true

console.log(reg.test('0')) // true
