# 手写观察者模式

```js
// 被观察对象
class Dep {
  constructor() {
    // 观察者列表
    this.subs = []
  }
  // 添加观察者
  addSubs(watcher) {
    if (watcher && watcher.update) {
      this.subs.push(watcher)
    }
  }
  // 事件发生时通知观察者进行处理
  notify() {
    this.subs.forEach((watcher) => {
      watcher.update()
    })
  }
}

// 观察者
class Watcher {
  constructor() {}
  // 事件发生时的处理方法
  update() {
    console.log('update')
  }
}

const dep = new Dep()
const watcher = new Watcher()

dep.addSubs(watcher)
dep.notify()
```
